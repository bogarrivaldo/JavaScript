-----------------------------------------------------------------------
# NAN (Not A Number)
- jika suatu bilangan dibagi string  [ 2 / 'valdo' = NAN ] 
- jika 0 dibagi 0 [ 0 / 0 = NAN ]
-----------------------------------------------------------------------
# - Infinity
jika bilangan utama minus dan dibagi 0 [-5 / 0 = -Infinity]
-----------------------------------------------------------------------
# Infinity
Jika suatu bilangan dibagi 0 akan infinity [ 5 / 0 = Infinity]
-----------------------------------------------------------------------

!!! Jika bilangan dibagi dengan string bilangan, hasilnya otomatis bilangan
contoh : 20 / '20' = 1 
nilai 1 akan otomatis menjadi int atau number
